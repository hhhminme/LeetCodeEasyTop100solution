const isPalindrome = (x) => {
  if (x < 0) {
    return false;
  }
  const palindrome = x.toString();
  const length = palindrome.length;

  for (let i = 0; i < length; i++) {
    let p = length - i - 1;
    if (p <= i) {
      break;
    }
    if (palindrome[i] !== palindrome[p]) {
      return false;
    }
  }
  return true;
};

const palindrome = (x) => {
  let reversed = x.toString().split("").reverse().join("");
  return x.toString() === reversed;
};
